{{/* pt-picker shortcode — 場景化選擇指南表格 用法： {{< pt-picker headers="你的需求|推薦選擇|原因" rows="追求高利率|樂天銀行|活存利率最高;常用LINE Pay|LINE Bank|LINE生態系整合好" >}} 參數： - headers: 表頭欄位，用 | 分隔 - rows: 資料列，每列用 ; 分隔，每欄用 | 分隔 */}} {{ $rawHeaders := .Get "headers" | default "" }} {{ $rawRows := .Get "rows" | default "" }} {{ $headers := split $rawHeaders "|" }} {{ $rows := split $rawRows ";" }}

{{ . }}
